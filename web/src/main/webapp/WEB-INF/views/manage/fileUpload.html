CREATE TABLE `T_EBK_AUTHOR` (
`FID` int(10) unsigned NOT NULL AUTO_INCREMENT COMMENT '作者ID',
`FNAME` varchar(64) NOT NULL DEFAULT '' COMMENT '作者名称',
`FICON` varchar(256) NOT NULL DEFAULT '' COMMENT '作者头像地址',
`FSUMMARY` varchar(1024) NOT NULL DEFAULT '' COMMENT '作者简介',
`FUPDATE_TIME` int(10) unsigned NOT NULL DEFAULT '0' COMMENT '更新时间',
`FCREATE_TIME` int(10) unsigned NOT NULL DEFAULT '0' COMMENT '创建时间',
`FREVISE_NAME` varchar(64) NOT NULL DEFAULT '' COMMENT '修改后的名称',
`FREVISE_SUMMARY` varchar(1024) NOT NULL DEFAULT '' COMMENT '修改后的简介',
PRIMARY KEY (`FID`),
UNIQUE KEY `UDX_NAME` (`FNAME`)
) ENGINE=InnoDB AUTO_INCREMENT=267308 DEFAULT CHARSET=utf8mb4 COMMENT='作者信息表'
